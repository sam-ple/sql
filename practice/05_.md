WITH member_increase_or_decrease AS (
SELECT
  *,
  CASE
    WHEN entry_flg = 0
      THEN -1
  ELSE 1
END AS change_num
FROM
  puzzle_1_table)
SELECT
  *,
  SUM(change_num) over(order by timestamp) AS member_num
FROM
  member_increase_or_decrease;
